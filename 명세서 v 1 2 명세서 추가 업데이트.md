function generateAssets(rockLevel, totalSprouts, totalFlowers) {

*// Lv.1: 지연된 에셋만*

if (rockLevel === 1) {
    return {
      sprouts: Array(totalSprouts).fill("adj_sprout"),

    };

    const guaranteed = Array(minAchievement).fill("ach_sprout");
    const random = Array(totalSprouts - minAchievement)
      .fill(0)
      .map(() => Math.random() < 0.7 ? "adj_sprout" : "ach_sprout");
    
    return {
      sprouts: [...guaranteed, ...random],
      flowers: Array(totalFlowers).fill("adj_flower")
    };
  }

*// Lv.3: 균형 (최소 각 40% 보장)*

else if (rockLevel === 3) {
    const minAdj = Math.floor(totalSprouts * 0.4);
    const minAch = Math.floor(totalSprouts * 0.4);
    const remaining = totalSprouts - minAdj - minAch;
    
    return {
      sprouts: [
        ...Array(minAdj).fill("adj_sprout"),
        ...Array(minAch).fill("ach_sprout"),
        ...Array(remaining).fill(Math.random() < 0.5 ? "adj_sprout" : "ach_sprout")
      ],
      flowers: [
        ...Array(Math.ceil(totalFlowers * 0.6)).fill("adj_flower"),
        ...Array(Math.floor(totalFlowers * 0.4)).fill("ach_flower")
      ]
    };
  }

*// Lv.4: 성공 우세 (최소 50% 보장)*

else if (rockLevel === 4) {
    const minAch = Math.floor(totalSprouts * 0.5);
    
    return {
      sprouts: [
        ...Array(minAch).fill("ach_sprout"),
        ...Array(totalSprouts - minAch)
          .fill(0)
          .map(() => Math.random() < 0.4 ? "adj_sprout" : "ach_sprout")
      ],
      flowers: [
        ...Array(Math.ceil(totalFlowers * 0.5)).fill("ach_flower"),
        ...Array(Math.floor(totalFlowers * 0.5)).fill("adj_flower")
      ]
    };
  }
}

function preloadImage(src) {
  return new Promise((resolve, reject) => {
